312 void
313 LaplaceProblem<dim, degree>::assemble_rhs()
314 {
315  if (settings.problem_type == "source")
316  {
317  TimerOutput::Scope timing(computing_timer, "Assemble right-hand side");
318 
319  MatrixFreeActiveVector solution_copy;
320  MatrixFreeActiveVector right_hand_side_copy;
321  stiffness_operator.initialize_dof_vector(solution_copy);
322  stiffness_operator.initialize_dof_vector(right_hand_side_copy);
323 
324  solution_copy = 0.;
325  constraints.distribute(solution_copy);
326  solution_copy.update_ghost_values();
327  right_hand_side_copy = 0;
328  const Table<2, VectorizedArray<double>> &coefficient =
329  *(stiffness_operator.get_coefficient());
330 
331  FEEvaluation<dim, degree, degree + 1, 1, double> phi(
332  *stiffness_operator.get_matrix_free());
333 
334  for (unsigned int cell = 0;
335  cell < stiffness_operator.get_matrix_free()->n_cell_batches();
336  ++cell)
337  {
338  phi.reinit(cell);
339  phi.read_dof_values_plain(solution_copy);
340  phi.evaluate(EvaluationFlags::gradients);
341 
342  for (unsigned int q = 0; q < phi.n_q_points; ++q)
343  {
344  phi.submit_gradient(-1.0 * (coefficient(cell, 0) *
345  phi.get_gradient(q)),
346  q);
347  phi.submit_value(
348  evaluate_function(settings.rhs, phi.quadrature_point(q), 0), q);
349  }
350 
351  phi.integrate_scatter(EvaluationFlags::values |
352  EvaluationFlags::gradients,
353  right_hand_side_copy);
354  }
355 
356  right_hand_side_copy.compress(VectorOperation::add);
357 
358  ChangeVectorTypes::copy(right_hand_side, right_hand_side_copy);
359  }

365 void
366 LaplaceProblem<dim, degree>::solve(const unsigned int cycle)
367 {
368  const bool is_intermediate_cycle = cycle > 0 && cycle < settings.n_cycles - 1;
369  bool same_controls = (settings.intermediate_solver_control.tolerance() ==
370  settings.first_and_last_solver_control.tolerance()) &&
371  (settings.intermediate_solver_control.max_steps() ==
372  settings.first_and_last_solver_control.max_steps()) &&
373  (settings.intermediate_solver_control.reduction() ==
374  settings.first_and_last_solver_control.reduction());
375 
376  if (settings.intermediate_solver_control.max_steps() == 0)
377  same_controls = true;
378 
379  SmartPointer<SolverControl> solver_control(
380  &settings.first_and_last_solver_control);
381 
382  if (is_intermediate_cycle && !same_controls)
383  solver_control = &settings.intermediate_solver_control;
384 
385  std::string section_name = ((is_intermediate_cycle) && !same_controls) ?
386  "Solve - intermediate" :
387  "Solve";
388 
389  TimerOutput::Scope timing(computing_timer, section_name);
390 
391  solution = 0.;
392  computing_timer.enter_subsection("Solve: Preconditioner setup");
393 
394  MGTransferMatrixFree<dim, float> mg_transfer(mg_constrained_dofs);
395  mg_transfer.build(dof_handler);
396 
397  SolverControl coarse_solver_control(1000, 1e-12, false, false);
398  SolverCG<MatrixFreeLevelVector> coarse_solver(coarse_solver_control);
399  PreconditionIdentity identity;
400  MGCoarseGridIterativeSolver<MatrixFreeLevelVector,
401  SolverCG<MatrixFreeLevelVector>,
402  MatrixFreeLevelMatrix,
403  PreconditionIdentity>
404  coarse_grid_solver(coarse_solver, mg_stiffness_operator[0], identity);
405 
406  using Smoother = dealii::PreconditionJacobi<MatrixFreeLevelMatrix>;
407  MGSmootherPrecondition<MatrixFreeLevelMatrix, Smoother, MatrixFreeLevelVector>
408  smoother;
409  smoother.initialize(mg_stiffness_operator,
410  typename Smoother::AdditionalData(
411  settings.gmg_smoother_dampen));
412  smoother.set_steps(settings.gmg_smoother_steps);
413 
414  mg::Matrix<MatrixFreeLevelVector> mg_m(mg_stiffness_operator);
415 
416  MGLevelObject<MatrixFreeOperators::MGInterfaceOperator<MatrixFreeLevelMatrix>>
417  mg_interface_matrices;
418  mg_interface_matrices.resize(0, triangulation.n_global_levels() - 1);
419  for (unsigned int level = 0; level < triangulation.n_global_levels(); ++level)
420  mg_interface_matrices[level].initialize(mg_stiffness_operator[level]);
421  mg::Matrix<MatrixFreeLevelVector> mg_interface(mg_interface_matrices);
422 
423  Multigrid<MatrixFreeLevelVector> mg(
424  mg_m, coarse_grid_solver, mg_transfer, smoother, smoother);
425  mg.set_edge_matrices(mg_interface, mg_interface);
426 
427  PreconditionMG<dim, MatrixFreeLevelVector, MGTransferMatrixFree<dim, float>>
428  preconditioner(dof_handler, mg, mg_transfer);
429 
430  // Copy the solution vector and right-hand side from LA::MPI::Vector
431  // to dealii::LinearAlgebra::distributed::Vector so that we can
432  // solve.
433  MatrixFreeActiveVector solution_copy;
434  MatrixFreeActiveVector right_hand_side_copy;
435  stiffness_operator.initialize_dof_vector(solution_copy);
436  stiffness_operator.initialize_dof_vector(right_hand_side_copy);
437 
438  ChangeVectorTypes::copy(solution_copy, solution);
439  ChangeVectorTypes::copy(right_hand_side_copy, right_hand_side);
440  computing_timer.leave_subsection("Solve: Preconditioner setup");
441 
442  // Timing for 1 V-cycle.
443  {
444  TimerOutput::Scope timing(computing_timer, "Solve: 1 multigrid V-cycle");
445  preconditioner.vmult(solution_copy, right_hand_side_copy);
446  }
447 
448  if (settings.problem_type == "source")
449  {
450  solution_copy = 0.;
451  // Solve the linear system, update the ghost values of the solution,
452  // copy back to LA::MPI::Vector and distribute constraints.
453  {
454  SolverCG<MatrixFreeActiveVector> solver(*solver_control);
455 
456  TimerOutput::Scope timing(computing_timer, "Solve: CG");
457  solver.solve(stiffness_operator,
458  solution_copy,
459  right_hand_side_copy,
460  preconditioner);
461  }
462 
463  solution_copy.update_ghost_values();
464  ChangeVectorTypes::copy(solution, solution_copy);
465  constraints.distribute(solution);
466  locally_relevant_solution = solution;
467 
468  pcout << " Number of CG iterations: "
469  << solver_control->last_step() << std::endl;
470  }
471  else if (settings.problem_type == "pinvit")
472  {
473  unsigned int current_pinvit_it = 0;
474  double current_error = 1e10;
475  std::vector<double> previous_eigenvalues = eigenvalues;
476 
477  do
478  {
479  TimerOutput::Scope timing(computing_timer, "Solve: pinvit steps");
480  one_step_pinvit(eigenvalues,
481  eigenvectors,
482  stiffness_operator,
483  mass_operator,
484  preconditioner,
485  constraints);
486  ++current_pinvit_it;
487  current_error = 0;
488  for (unsigned int i = 0; i < eigenvalues.size(); ++i)
489  current_error +=
490  std::abs(eigenvalues[i] - previous_eigenvalues[i]) /
491  eigenvalues[i];
492  current_error /= eigenvalues.size();
493  previous_eigenvalues = eigenvalues;
494  }
495  while (solver_control->check(current_pinvit_it, current_error) ==
496  SolverControl::iterate);
497 
498  pcout << std::endl
499  << " Error(" << current_pinvit_it << ") = " << current_error
500  << std::endl;
501  pcout << " Eigenvalues(" << current_pinvit_it
502  << ") = " << Patterns::Tools::to_string(eigenvalues) << std::endl;
503 
504  for (auto &v : eigenvectors)
505  v.update_ghost_values();
506 
507  pcout << " Number of PINVIT iterations: "
508  << solver_control->last_step() << std::endl;
509  }
510 }

575 template <int dim, int degree>
576 void
577 LaplaceProblem<dim, degree>::estimate()
578 {
579  TimerOutput::Scope timing(computing_timer, "Estimate");
580 
581  if (settings.problem_type == "source")
582  {
583  locally_relevant_solution = solution;
584 
585  estimated_error_square_per_cell.reinit(triangulation.n_active_cells());
586 
587  using Iterator = typename DoFHandler<dim>::active_cell_iterator;
588 
589  // Assembler for cell residual $h^2 \| f + \epsilon \triangle u \|_K^2$
590  auto cell_worker = [&](const Iterator & cell,
591  ScratchData<dim> &scratch_data,
592  CopyData & copy_data) {
593  FEValues<dim> &fe_values = scratch_data.fe_values;
594  fe_values.reinit(cell);
595 
596  const double rhs_value = settings.rhs.value(cell->center());
597 
598  const double nu = settings.coefficient.value(cell->center());
599 
600  std::vector<Tensor<2, dim>> hessians(fe_values.n_quadrature_points);
601  fe_values.get_function_hessians(locally_relevant_solution, hessians);
602 
603  copy_data.cell_index = cell->active_cell_index();
604 
605  double residual_norm_square = 0.;
606  for (unsigned k = 0; k < fe_values.n_quadrature_points; ++k)
607  {
608  const double residual = (rhs_value + nu * trace(hessians[k]));
609  residual_norm_square += residual * residual * fe_values.JxW(k);
610  }
611 
612  copy_data.value =
613  cell->diameter() * cell->diameter() * residual_norm_square;
614  };
615 
616  // Assembler for face term $\sum_F h_F \| \jump{\epsilon \nabla u \cdot n}
617  // \|_F^2$
618  auto face_worker = [&](const Iterator & cell,
619  const unsigned int &f,
620  const unsigned int &sf,
621  const Iterator & ncell,
622  const unsigned int &nf,
623  const unsigned int &nsf,
624  ScratchData<dim> & scratch_data,
625  CopyData & copy_data) {
626  FEInterfaceValues<dim> &fe_interface_values =
627  scratch_data.fe_interface_values;
628  fe_interface_values.reinit(cell, f, sf, ncell, nf, nsf);
629 
630  copy_data.face_data.emplace_back();
631  CopyData::FaceData &copy_data_face = copy_data.face_data.back();
632 
633  copy_data_face.cell_indices[0] = cell->active_cell_index();
634  copy_data_face.cell_indices[1] = ncell->active_cell_index();
635 
636  const double coeff1 = settings.coefficient.value(cell->center());
637  const double coeff2 = settings.coefficient.value(ncell->center());
638 
639  std::vector<Tensor<1, dim>> grad_u[2];
640 
641  for (unsigned int i = 0; i < 2; ++i)
642  {
643  grad_u[i].resize(fe_interface_values.n_quadrature_points);
644  fe_interface_values.get_fe_face_values(i).get_function_gradients(
645  locally_relevant_solution, grad_u[i]);
646  }
647 
648  double jump_norm_square = 0.;
649 
650  for (unsigned int qpoint = 0;
651  qpoint < fe_interface_values.n_quadrature_points;
652  ++qpoint)
653  {
654  const double jump =
655  coeff1 * grad_u[0][qpoint] * fe_interface_values.normal(qpoint) -
656  coeff2 * grad_u[1][qpoint] * fe_interface_values.normal(qpoint);
657 
658  jump_norm_square += jump * jump * fe_interface_values.JxW(qpoint);
659  }
660 
661  const double h = cell->face(f)->measure();
662  copy_data_face.values[0] = 0.5 * h * jump_norm_square;
663  copy_data_face.values[1] = copy_data_face.values[0];
664  };
665 
666  auto copier = [&](const CopyData &copy_data) {
667  if (copy_data.cell_index != numbers::invalid_unsigned_int)
668  estimated_error_square_per_cell[copy_data.cell_index] +=
669  copy_data.value;
670 
671  for (auto &cdf : copy_data.face_data)
672  for (unsigned int j = 0; j < 2; ++j)
673  estimated_error_square_per_cell[cdf.cell_indices[j]] +=
674  cdf.values[j];
675  };
676 
677  const unsigned int n_gauss_points = degree + 1;
678  ScratchData<dim> scratch_data(mapping,
679  fe,
680  n_gauss_points,
681  update_hessians | update_quadrature_points |
682  update_JxW_values | update_values,
683  update_values | update_gradients |
684  update_JxW_values |
685  update_normal_vectors);
686  CopyData copy_data;
687 
688  // We need to assemble each interior face once but we need to make sure
689  // that both processes assemble the face term between a locally owned and
690  // a ghost cell. This is achieved by setting the
691  // MeshWorker::assemble_ghost_faces_both flag. We need to do this, because
692  // we do not communicate the error estimator contributions here.
693  MeshWorker::mesh_loop(dof_handler.begin_active(),
694  dof_handler.end(),
695  cell_worker,
696  copier,
697  scratch_data,
698  copy_data,
699  MeshWorker::assemble_own_cells |
700  MeshWorker::assemble_ghost_faces_both |
701  MeshWorker::assemble_own_interior_faces_once,
702  /*boundary_worker=*/nullptr,
703  face_worker);
704 
705  const double global_error_estimate =
706  std::sqrt(Utilities::MPI::sum(estimated_error_square_per_cell.l1_norm(),
707  mpi_communicator));
708  pcout << " Global error estimate: " << global_error_estimate
709  << std::endl;
710  }
711  else
712  {
713  estimated_error_square_per_cell.reinit(triangulation.n_active_cells());
714  for (const auto eigenvector_index : settings.eigen_estimators)
715  if (eigenvector_index < settings.number_of_eigenvalues)
716  {
717  pcout << " Computing estimator on eigenvector "
718  << eigenvector_index << std::endl;
719  const auto lambda = eigenvalues[eigenvector_index];
720  ChangeVectorTypes::copy(locally_relevant_solution,
721  eigenvectors[eigenvector_index]);
722 
723  using Iterator = typename DoFHandler<dim>::active_cell_iterator;
724 
725  // Assembler for cell residual $h^2 \| f + \epsilon \triangle u
726  // \|_K^2$
727  auto cell_worker = [&](const Iterator & cell,
728  ScratchData<dim> &scratch_data,
729  CopyData & copy_data) {
730  FEValues<dim> &fe_values = scratch_data.fe_values;
731  fe_values.reinit(cell);
732 
733  const double nu = settings.coefficient.value(cell->center());
734 
735  std::vector<Tensor<2, dim>> hessians(
736  fe_values.n_quadrature_points);
737  fe_values.get_function_hessians(locally_relevant_solution,
738  hessians);
739 
740  std::vector<double> values(fe_values.n_quadrature_points);
741  fe_values.get_function_values(locally_relevant_solution, values);
742 
743  copy_data.cell_index = cell->active_cell_index();
744 
745  double residual_norm_square = 0.;
746  for (unsigned k = 0; k < fe_values.n_quadrature_points; ++k)
747  {
748  const double residual =
749  (lambda * values[k] + nu * trace(hessians[k]));
750  residual_norm_square +=
751  residual * residual * fe_values.JxW(k);
752  }
753 
754  copy_data.value =
755  cell->diameter() * cell->diameter() * residual_norm_square;
756  };
757 
758  // Assembler for face term $\sum_F h_F \| \jump{\epsilon \nabla u
759  // \cdot n}
760  // \|_F^2$
761  auto face_worker = [&](const Iterator & cell,
762  const unsigned int &f,
763  const unsigned int &sf,
764  const Iterator & ncell,
765  const unsigned int &nf,
766  const unsigned int &nsf,
767  ScratchData<dim> & scratch_data,
768  CopyData & copy_data) {
769  FEInterfaceValues<dim> &fe_interface_values =
770  scratch_data.fe_interface_values;
771  fe_interface_values.reinit(cell, f, sf, ncell, nf, nsf);
772 
773  copy_data.face_data.emplace_back();
774  CopyData::FaceData &copy_data_face = copy_data.face_data.back();
775 
776  copy_data_face.cell_indices[0] = cell->active_cell_index();
777  copy_data_face.cell_indices[1] = ncell->active_cell_index();
778 
779  const double coeff1 = settings.coefficient.value(cell->center());
780  const double coeff2 = settings.coefficient.value(ncell->center());
781 
782  std::vector<Tensor<1, dim>> grad_u[2];
783 
784  for (unsigned int i = 0; i < 2; ++i)
785  {
786  grad_u[i].resize(fe_interface_values.n_quadrature_points);
787  fe_interface_values.get_fe_face_values(i)
788  .get_function_gradients(locally_relevant_solution,
789  grad_u[i]);
790  }
791 
792  double jump_norm_square = 0.;
793 
794  for (unsigned int qpoint = 0;
795  qpoint < fe_interface_values.n_quadrature_points;
796  ++qpoint)
797  {
798  const double jump = coeff1 * grad_u[0][qpoint] *
799  fe_interface_values.normal(qpoint) -
800  coeff2 * grad_u[1][qpoint] *
801  fe_interface_values.normal(qpoint);
802 
803  jump_norm_square +=
804  jump * jump * fe_interface_values.JxW(qpoint);
805  }
806 
807  const double h = cell->face(f)->measure();
808  copy_data_face.values[0] = 0.5 * h * jump_norm_square;
809  copy_data_face.values[1] = copy_data_face.values[0];
810  };
811 
812  auto copier = [&](const CopyData &copy_data) {
813  if (copy_data.cell_index != numbers::invalid_unsigned_int)
814  estimated_error_square_per_cell[copy_data.cell_index] +=
815  copy_data.value;
816 
817  for (auto &cdf : copy_data.face_data)
818  for (unsigned int j = 0; j < 2; ++j)
819  estimated_error_square_per_cell[cdf.cell_indices[j]] +=
820  cdf.values[j];
821  };
822 
823  const unsigned int n_gauss_points = degree + 1;
824  ScratchData<dim> scratch_data(mapping,
825  fe,
826  n_gauss_points,
827  update_hessians | update_JxW_values |
828  update_values,
829  update_values | update_gradients |
830  update_normal_vectors);
831  CopyData copy_data;
832 
833  // We need to assemble each interior face once but we need to make
834  // sure that both processes assemble the face term between a locally
835  // owned and a ghost cell. This is achieved by setting the
836  // MeshWorker::assemble_ghost_faces_both flag. We need to do this,
837  // because we do not communicate the error estimator contributions
838  // here.
839  MeshWorker::mesh_loop(
840  dof_handler.begin_active(),
841  dof_handler.end(),
842  cell_worker,
843  copier,
844  scratch_data,
845  copy_data,
846  MeshWorker::assemble_own_cells |
847  MeshWorker::assemble_ghost_faces_both |
848  MeshWorker::assemble_own_interior_faces_once,
849  /*boundary_worker=*/nullptr,
850  face_worker);
851  }
852  const double global_error_estimate =
853  std::sqrt(Utilities::MPI::sum(estimated_error_square_per_cell.l1_norm(),
854  mpi_communicator));

856  << std::endl;

862 template <int dim, int degree>
863 void
864 LaplaceProblem<dim, degree>::refine_grid()
865 {
866  TimerOutput::Scope timing(computing_timer, "Refine grid");
867 
868  const double refinement_fraction = 1. / (std::pow(2.0, dim) - 1.);
869  if (settings.refinement_strategy == "fixed_number")
870  parallel::distributed::GridRefinement::refine_and_coarsen_fixed_number(
871  triangulation, estimated_error_square_per_cell, refinement_fraction, 0.0);
872  else if (settings.refinement_strategy == "fixed_fraction")
873  parallel::distributed::GridRefinement::refine_and_coarsen_fixed_fraction(
874  triangulation, estimated_error_square_per_cell, refinement_fraction, 0.0);
875  else if (settings.refinement_strategy == "global")
876  for (const auto &cell : triangulation.active_cell_iterators())
877  cell->set_refine_flag();
878  else
879  Assert(false, ExcInternalError("Unknown refinement strategy."));
880 
881  if (settings.problem_type == "pinvit")
882  {
883  TimerOutput::Scope timing(computing_timer, "Transfer: prepare");
884  std::vector<const MatrixFreeActiveVector *> all_in;
885  for (const auto &v : eigenvectors)
886  all_in.push_back(&v);
887  eigenvectors_transfer.prepare_for_coarsening_and_refinement(all_in);
888  }
889  else
890  {
891  TimerOutput::Scope timing(computing_timer, "Transfer: prepare");
892  solution_transfer.prepare_for_coarsening_and_refinement(
893  locally_relevant_solution);
894  }
895  triangulation.execute_coarsening_and_refinement();
896  print_grid_info();
897 }
