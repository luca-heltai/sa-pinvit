32 template <int dim, typename Number>
33 VectorizedArray<Number>
34 evaluate_function(const Function<dim> & function,
35  const Point<dim, VectorizedArray<Number>> &p_vectorized,
36  const unsigned int component)
37 {
38  VectorizedArray<Number> result;
39  for (unsigned int v = 0; v < p_vectorized[0].size(); ++v)
40  {
41  Point<dim> p;
42  for (unsigned int d = 0; d < dim; ++d)
43  p[d] = p_vectorized[d][v];
44  result[v] = function.value(p, component);
45  }
46  return result;
47 }

70 // operations which copy to and from Trilinos vectors for compatibility with

115 template <int dim, typename number>
116 std::shared_ptr<Table<2, VectorizedArray<number>>>
117 make_coefficient_table(
118  const Function<dim> & function,
119  const MatrixFree<dim, number, VectorizedArray<number>> &mf_storage)
120 {
121  auto coefficient_table =
122  std::make_shared<Table<2, VectorizedArray<number>>>();
123  FEEvaluation<dim, -1, 0, 1, number> fe_eval(mf_storage);
124  const unsigned int n_cells = mf_storage.n_macro_cells();
125  const unsigned int n_q_points = fe_eval.n_q_points;
126  coefficient_table->reinit(n_cells, 1);
127  for (unsigned int cell = 0; cell < n_cells; ++cell)
128  {
129  fe_eval.reinit(cell);
130  VectorizedArray<number> average_value = 0.;
131  for (unsigned int q = 0; q < n_q_points; ++q)
132  average_value +=
133  evaluate_function(function, fe_eval.quadrature_point(q), 0);
134  average_value /= n_q_points;
135  (*coefficient_table)(cell, 0) = average_value;
136  }
137  return coefficient_table;
138 }
