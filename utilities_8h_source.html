106 template <int dim, typename number>
107 std::shared_ptr<Table<2, VectorizedArray<number>>>
108 make_coefficient_table(
109  const Function<dim> & function,
110  const MatrixFree<dim, number, VectorizedArray<number>> &mf_storage)
111 {
112  auto coefficient_table =
113  std::make_shared<Table<2, VectorizedArray<number>>>();
114  FEEvaluation<dim, -1, 0, 1, number> fe_eval(mf_storage);
115  const unsigned int n_cells = mf_storage.n_cell_batches();
116  const unsigned int n_q_points = fe_eval.n_q_points;
117  coefficient_table->reinit(n_cells, 1);
118  for (unsigned int cell = 0; cell < n_cells; ++cell)
119  {
120  fe_eval.reinit(cell);
121  VectorizedArray<number> average_value = 0.;
122  for (unsigned int q = 0; q < n_q_points; ++q)
123  average_value +=
124  evaluate_function(function, fe_eval.quadrature_point(q), 0);
125  average_value /= n_q_points;
126  (*coefficient_table)(cell, 0) = average_value;
127  }
128  return coefficient_table;
129 }

132 template <int dim, int spacedim>
133 void
134 read_grid_and_cad_files(const std::string & grid_file_name,
135  const std::string & ids_and_cad_file_names,
136  Triangulation<dim, spacedim> &tria)
137 {
138  GridIn<dim, spacedim> grid_in;
139  grid_in.attach_triangulation(tria);
140  grid_in.read(grid_file_name);
141 
142  // If we got to this point, then the Triangulation has been read, and we are
143  // ready to attach to it the correct manifold descriptions. We perform the
144  // next lines of code only if deal.II has been built with OpenCASCADE
145  // support. For each entry in the map, we try to open the corresponding CAD
146  // file, we analyze it, and according to its content, opt for either a
147  // OpenCASCADE::ArcLengthProjectionLineManifold (if the CAD file contains a
148  // single `TopoDS_Edge` or a single `TopoDS_Wire`) or a
149  // OpenCASCADE::NURBSPatchManifold, if the file contains a single face.
150  // Notice that if the CAD files do not contain single wires, edges, or
151  // faces, an assertion will be throw in the generation of the Manifold.
152  //
153  // We use the Patterns::Tools::Convert class to do the conversion from the
154  // string to a map between manifold ids and file names for us:
155 #ifdef DEAL_II_WITH_OPENCASCADE
156  using map_type = std::map<types::manifold_id, std::string>;
157  using Converter = Patterns::Tools::Convert<map_type>;
158 
159  for (const auto &pair : Converter::to_value(ids_and_cad_file_names))
160  {
161  const auto &manifold_id = pair.first;
162  const auto &cad_file_name = pair.second;
163 
164  const auto extension = boost::algorithm::to_lower_copy(
165  cad_file_name.substr(cad_file_name.find_last_of('.') + 1));
166 
167  TopoDS_Shape shape;
168  if (extension == "iges" || extension == "igs")
169  shape = OpenCASCADE::read_IGES(cad_file_name);
170  else if (extension == "step" || extension == "stp")
171  shape = OpenCASCADE::read_STEP(cad_file_name);
172  else
173  AssertThrow(false,
174  ExcNotImplemented("We found an extension that we "
175  "do not recognize as a CAD file "
176  "extension. Bailing out."));
177 
178  // Now we check how many faces are contained in the `Shape`. OpenCASCADE
179  // is intrinsically 3D, so if this number is zero, we interpret this as
180  // a line manifold, otherwise as a
181  // OpenCASCADE::NormalToMeshProjectionManifold in `spacedim` = 3, or
182  // OpenCASCADE::NURBSPatchManifold in `spacedim` = 2.
183  const auto n_elements = OpenCASCADE::count_elements(shape);
184  if ((std::get<0>(n_elements) == 0))
185  tria.set_manifold(
186  manifold_id,
187  OpenCASCADE::ArclengthProjectionLineManifold<dim, spacedim>(shape));
188  else if (spacedim == 3)
189  {
190  // We use this trick, because
191  // OpenCASCADE::NormalToMeshProjectionManifold is only implemented
192  // for spacedim = 3. The check above makes sure that things actually
193  // work correctly.
194  const auto t = reinterpret_cast<Triangulation<dim, 3> *>(&tria);
195  t->set_manifold(manifold_id,
196  OpenCASCADE::NormalToMeshProjectionManifold<dim, 3>(
197  shape));
198  }
199  else
200  // We also allow surface descriptions in two dimensional spaces based
201  // on single NURBS patches. For this to work, the CAD file must
202  // contain a single `TopoDS_Face`.
203  tria.set_manifold(manifold_id,
204  OpenCASCADE::NURBSPatchManifold<dim, spacedim>(
205  TopoDS::Face(shape)));
206  }
207 #else
208  (void)ids_and_cad_file_names;
209  AssertThrow(false, ExcNotImplemented("Generation of the grid failed."));
210 #endif
211 }
